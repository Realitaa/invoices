Enum accepted {
  "Sudah"
  "Belum"
  "Cancel"
}

Enum reason {
  "PROSES AOSODOMORO"
  "RENEWAL KONTRAK"
  "ADJUSTMENT"
  "BUNDLING"
  "BY REKON/USAGE"
}

Table invoice {
  id string [primary key] // manual
  customer_id string
  amount integer
  reason reason
  payment_status boolean
  periode date
  // bank & va merupakan kombinasi va dan id
  due_date date
  accepted_invoice accepted
  delayed_paying_reason text
  created_at timestamp
}

// check
Table customer {
  id string [primary key] // manual, nomor customer
  name string
  address text
  npwp string
  created_at timestamp
}
