Enum accepted {
  "Sudah"
  "Belum"
  "Cancel"
}

Enum reason {
  "PROSES AOSODOMORO"
  "RENEWAL KONTRAK"
  "ADJUSTMENT"
  "BUNDLING"
  "BY REKON/USAGE"
}

Table invoice {
  id string [primary key] // manual
  customer_id uuid
  reason reason
  payment_status boolean
  month_periode tinyint
  year_periode int
  // bank & va merupakan kombinasi va dan id
  due_date date
  accepted_invoice accepted
  // apakah ini dibutuhkan?
  delayed_paying_reason text
  created_at timestamp
}

// mengambil data dari existing tables
Table customers {
  id uuid [primary key]
  id_number varchar
  account_name varchar
  npwp_trems varchar
  address text
  ubis varchar
  bisnis_area varchar
  business_share varchar
  divisi varchar
  witel varchar
  created_at timestamp
}

// Kandidat table
Table invoice_reason {
  id integer [primary key]
  invoice_id string
  manual_invoice_reason text
  order_number string
  last_status text
  commitment date
}

// Tabel daftar service pada invoices
Table services {
  id integer [primary key]
  invoice_id uuid
  name varchar
}

// Table untuk menyimpan daftar services
Table service_items {
  id integer [primary key]
  service_id uuid
  description text
  service_items_id varchar
  bandwith varchar
  year_periode integer
  month_periode tinyint
  amount bigint
}

Ref account_id: customers.id < invoice.customer_id

Ref: "invoice_reason"."invoice_id" < "invoice"."id"

Ref: "services"."invoice_id" < "invoice"."id"

Ref: "service_items"."service_id" < "services"."id"
